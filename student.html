function doPost(e) { if (!e || !e.parameter) { return ContentService.createTextOutput("Error: No data received"); } var sheet = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action; // TEACHER CREATES SESSION if (action == "create_session") { var ws = sheet.getSheetByName("Sessions"); ws.appendRow([ e.parameter.code, e.parameter.ip, new Date() ]); return ContentService.createTextOutput("Session Created"); } // STUDENT SUBMITS ATTENDANCE if (action == "submit_attendance") { var sessionSheet = sheet.getSheetByName("Sessions"); var sessionData = sessionSheet.getDataRange().getValues(); var teacherIP = ""; // Find session code for (var i = sessionData.length - 1; i >= 0; i--) { if (sessionData[i][0] == e.parameter.code) { teacherIP = sessionData[i][1]; break; } } if (teacherIP == "") { return ContentService.createTextOutput("Error: Invalid Room Code"); } // WiFi Check if (teacherIP != e.parameter.ip) { return ContentService.createTextOutput("Error: Connect to Classroom WiFi"); } var attSheet = sheet.getSheetByName("Attendance"); var attendanceData = attSheet.getDataRange().getValues(); var name = e.parameter.name; var batch = e.parameter.batch; var date = new Date(); var dateStr = Utilities.formatDate( date, Session.getScriptTimeZone(), "yyyy-MM-dd" ); var timeStr = Utilities.formatDate( date, Session.getScriptTimeZone(), "HH:mm:ss" ); // DUPLICATE CHECK for (var j = 1; j < attendanceData.length; j++) { var existingDate = attendanceData[j][0]; var existingName = attendanceData[j][2]; var existingBatch = attendanceData[j][3]; if ( existingDate == dateStr && existingName == name && existingBatch == batch ) { return ContentService.createTextOutput("Error: Attendance already marked"); } } // SAVE ATTENDANCE attSheet.appendRow([ dateStr, timeStr, name, batch ]); return ContentService.createTextOutput("Success"); } }
